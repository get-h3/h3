01 / Architecture

How It Works

H3 separates the body from the brain. Hermes owns the platform surface; your harness owns the reasoning loop.

Hermes is the bodyChannels, credentials, tools, and delivery stay put.
H3 is the neural linkTwo core callbacks move decisions across the boundary.
Your harness is the brainSwap agent frameworks without rebuilding the platform.

02 / The idea

What is H3?

A narrow interface for a wide ecosystem of agent runtimes.

Bring the brain you already trust.

H3, short for Hermes Harness Hooks, is a protocol for connecting an external agent harness to Hermes. Hermes receives the user message, executes the decisions your harness returns, and sends each result back for the next step.

That means a CrewAI workflow, a LangChain graph, a custom Go loop, or a research agent can become the thinking layer without taking on transport, channel integrations, or platform operations.

The wire format stays the same across languages. The implementation stays yours.

03 / Quickstart

From zero to echo in minutes.

Pick a language, install its SDK, and start with a harness that responds to /v1/process and closes on /v1/result.

Each SDK wraps the protocol types and HTTP routes so your code can focus on decisions. Copy the example, run it on port 9191, then point the test battery at it.

Go · minimal echo harnessgo get github.com/get-h3/sdk-go
package main

import (
  "fmt"
  "net/http"
  "github.com/get-h3/sdk-go/harness"
  "github.com/get-h3/sdk-go/protocol"
)

type Echo struct{}
func (Echo) OnProcess(r *protocol.ProcessRequest) (*protocol.Decision, error) {
  return &protocol.Decision{Decision: protocol.DecisionText,
    DecisionID: "echo-1", Text: &protocol.TextResp{Content: fmt.Sprintf("Echo: %s", r.Message.Content), Finished: true}}, nil
}
func (Echo) OnResult(*protocol.ResultRequest) (*protocol.Decision, error) {
  return &protocol.Decision{Decision: protocol.DecisionEnd, DecisionID: "done", End: &protocol.End{Reason: protocol.EndTaskComplete}}, nil
}
func (Echo) OnCancel(*protocol.CancelRequest) error { return nil }
func (Echo) OnSessionTerminate(string) error { return nil }
func (Echo) Health() *protocol.HealthResponse { return &protocol.HealthResponse{Status: protocol.HealthOK, Version: "1.0.0", Transport: "rest", ProtocolVersion: "1.0"} }
func main() { fmt.Println("H3 harness listening on :9191"); http.ListenAndServe(":9191", harness.NewHTTPServer(Echo{})) }

Python · FastAPI echo harnesspip install h3-harness-sdk fastapi uvicorn
from fastapi import FastAPI
from h3_harness import BaseHarness, Decision, DecisionType, End, TextResponse, create_router

class Echo(BaseHarness):
    async def on_process(self, req):
        return Decision(decision=DecisionType.TEXT, text=TextResponse(content=f"Echo: {req.message.content}", finished=True))

    async def on_result(self, req):
        return Decision(decision=DecisionType.END, end=End(reason="task_complete"))

app = FastAPI()
app.include_router(create_router(Echo()))
Run:uvicorn harness:app --host 0.0.0.0 --port 9191
TypeScript · Hono echo harnessnpm install @get-h3/h3-harness-sdk hono
import { Hono } from "hono";
import { createH3Router } from "@get-h3/h3-harness-sdk";
import type { Harness, Decision } from "@get-h3/h3-harness-sdk";

const echo: Harness = {
  async onProcess(req) { return { decision: "text", decision_id: crypto.randomUUID(), text: { content: `Echo: ${req.message.content}`, finished: true } }; },
  async onResult(_req) { return { decision: "end", decision_id: crypto.randomUUID(), end: { reason: "task_complete" } }; },
  health() { return { status: "ok", version: "1.0.0", transport: "rest", protocol_version: "1.0", capabilities: ["text", "end"] }; }
};
const app = new Hono(); app.route("/", createH3Router(echo)); export default app;
